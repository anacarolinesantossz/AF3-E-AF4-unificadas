      // Bubble Sort — implementado do zero, sem usar .sort()
//
// Como funciona:
//  1. O array é percorrido várias vezes ("passagens").
//  2. Em cada passagem, pares de elementos ADJACENTES são comparados.
//  3. Se o da esquerda for MAIOR que o da direita → os dois são TROCADOS.
//  4. Ao final de cada passagem, o maior elemento "borbulha" para o fim.
//  5. Repete até que uma passagem inteira não produza nenhuma troca.
//  Complexidade: O(n²) — adequado para arrays pequenos.

function bubbleSort(arr) {
  const n = arr.length;

  for (let passagem = 0; passagem < n - 1; passagem++) {
    let houveTroca = false;

    for (let j = 0; j < n - 1 - passagem; j++) {
      if (arr[j].preco > arr[j + 1].preco) {
        // troca os dois elementos adjacentes
        const temp = arr[j];
        arr[j]     = arr[j + 1];
        arr[j + 1] = temp;
        houveTroca = true;
      }
    }

    if (!houveTroca) break; // nenhuma troca → array já ordenado
  }

  return arr;
}
